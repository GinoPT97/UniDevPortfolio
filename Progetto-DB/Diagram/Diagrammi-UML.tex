\section*{Diagramma non ristrutturato}

\begin{center}
\begin{tikzpicture}[node distance=3cm]

% Persona
\node[class] (persona) {
  \textbf{Persona}\\[-0.3em]\rule{\linewidth}{0.2pt}\\
  + nome : string\\
  + cognome : string\\
  + codicefiscale : string\\
  + indirizzo : string\\
  + telefono : string\\
  + email : string
};

% Cliente
\node[class, below left=of persona] (cliente) {
  \textbf{Cliente}\\[-0.3em]\rule{\linewidth}{0.2pt}\\
  + codcliente : int
};

% Dipendente
\node[class, below right=of persona] (dipendente) {
  \textbf{Dipendente}\\[-0.3em]\rule{\linewidth}{0.2pt}\\
  + coddipendente : int
};

% Tessera
\node[class, right=4cm of cliente] (tessera) {
  \textbf{Tessera}\\[-0.3em]\rule{\linewidth}{0.2pt}\\
  + codtessera : int\\
  + numeropunti : numeric\\
  + dataemissione : date\\
  + datascadenza : date\\
  + stato : {ATTIVA, SOSPESA, SCADUTA}
};

% Ordine
\node[class, below=5cm of persona] (ordine) {
  \textbf{Ordine}\\[-0.3em]\rule{\linewidth}{0.2pt}\\
  + codordine : int\\
  + prezzototale : real\\
  + dataacquisto : date
};

% ArticoliOrdine
\node[class, right=5cm of ordine] (articoli) {
  \textbf{ArticoliOrdine}\\[-0.3em]\rule{\linewidth}{0.2pt}\\
  + prezzo : numeric\\
  + numeroarticoli : int
};

% Prodotto
\node[class, right=5cm of tessera] (prodotto) {
  \textbf{Prodotto}\\[-0.3em]\rule{\linewidth}{0.2pt}\\
  + codprodotto : int\\
  + nome : string\\
  + descrizione : string\\
  + prezzo : numeric\\
  + luogoprovenienza : string\\
  + scorta : int
};

% Specializzazioni di Prodotto
\node[class, below left=of prodotto] (frutta) {
  \textbf{Frutta}\\[-0.3em]\rule{\linewidth}{0.2pt}\\
  + dataraccolta : date
};
\node[class, below=of prodotto] (verdura) {
  \textbf{Verdura}\\[-0.3em]\rule{\linewidth}{0.2pt}\\
  + dataraccolta : date
};
\node[class, below right=of prodotto] (latticini) {
  \textbf{Latticini}\\[-0.3em]\rule{\linewidth}{0.2pt}\\
  + datamungitura : date\\
  + dataproduzione : date\\
  + datascadenza : date
};
\node[class, right=5cm of latticini] (uova) {
  \textbf{Uova}\\[-0.3em]\rule{\linewidth}{0.2pt}\\
  + datascadenza : date
};
\node[class, left=5cm of latticini] (farinacei) {
  \textbf{Farinacei}\\[-0.3em]\rule{\linewidth}{0.2pt}\\
  + glutine : boolean
};
\node[class, below=5cm of verdura] (confezionati) {
  \textbf{Confezionati}\\[-0.3em]\rule{\linewidth}{0.2pt}\\
  + datascadenza : date
};

% Relazioni
\draw[relation] (persona.south west) -- (cliente.north);
\draw[relation] (persona.south east) -- (dipendente.north);
\draw[relation] (cliente) -- (tessera);
\draw[relation] (cliente) -- (ordine);
\draw[relation] (dipendente) -- (ordine);
\draw[relation] (ordine) -- (articoli);
\draw[relation] (prodotto) -- (articoli);

% Generalizzazione prodotto
\draw[relation] (prodotto.south) -- (frutta.north);
\draw[relation] (prodotto.south) -- (verdura.north);
\draw[relation] (prodotto.south) -- (latticini.north);
\draw[relation] (prodotto.south) -- (uova.north);
\draw[relation] (prodotto.south) -- (farinacei.north);
\draw[relation] (prodotto.south) -- (confezionati.north);

\end{tikzpicture}
\end{center}
