\section*{Diagramma non ristrutturato}

\begin{center}
\begin{tikzpicture}[node distance=3.2cm and 5.2cm]

% Livello superiore - Persona come entità principale (centrale)
\node[class] (persona) at (0,0) {
  \textbf{Persona}\\[-0.3em]\rule{\linewidth}{0.2pt}\\
  + nome : string\\
  + cognome : string\\
  + codicefiscale : string\\
  + indirizzo : string\\
  + telefono : string\\
  + email : string
};

% Livello secondo - Specializzazioni di Persona (più distanziate)
\node[class] (cliente) at (-7,-3.5) {
  \textbf{Cliente}\\[-0.3em]\rule{\linewidth}{0.2pt}\\
  + codcliente : int
};

\node[class] (dipendente) at (7,-3.5) {
  \textbf{Dipendente}\\[-0.3em]\rule{\linewidth}{0.2pt}\\
  + coddipendente : int
};

% Livello terzo - Entità collegate a Cliente (allineate sotto cliente)
\node[class] (tessera) at (-7,-8) {
  \textbf{Tessera}\\[-0.3em]\rule{\linewidth}{0.2pt}\\
  + codtessera : int\\
  + numeropunti : numeric\\
  + dataemissione : date\\
  + datascadenza : date\\
  + stato : {ATTIVA, SOSPESA, SCADUTA}
};

% Livello centrale - Ordine come entità di collegamento
\node[class] (ordine) at (0,-8) {
  \textbf{Ordine}\\[-0.3em]\rule{\linewidth}{0.2pt}\\
  + codordine : int\\
  + prezzototale : real\\
  + dataacquisto : date
};

% Livello quarto - ArticoliOrdine
\node[class] (articoli) at (0,-12) {
  \textbf{ArticoliOrdine}\\[-0.3em]\rule{\linewidth}{0.2pt}\\
  + prezzo : numeric\\
  + numeroarticoli : int
};

% Livello destro - Prodotto e sue specializzazioni (spostato più a destra)
\node[class] (prodotto) at (12,-8) {
  \textbf{Prodotto}\\[-0.3em]\rule{\linewidth}{0.2pt}\\
  + codprodotto : int\\
  + nome : string\\
  + descrizione : string\\
  + prezzo : numeric\\
  + luogoprovenienza : string\\
  + scorta : int
};

% Specializzazioni di Prodotto - disposte simmetricamente
\node[class] (frutta) at (2,-15) {
  \textbf{Frutta}\\[-0.3em]\rule{\linewidth}{0.2pt}\\
  + dataraccolta : date
};

\node[class] (verdura) at (5,-15) {
  \textbf{Verdura}\\[-0.3em]\rule{\linewidth}{0.2pt}\\
  + dataraccolta : date
};

\node[class] (latticini) at (8,-15) {
  \textbf{Latticini}\\[-0.3em]\rule{\linewidth}{0.2pt}\\
  + datamungitura : date\\
  + dataproduzione : date\\
  + datascadenza : date
};

\node[class] (farinacei) at (11,-15) {
  \textbf{Farinacei}\\[-0.3em]\rule{\linewidth}{0.2pt}\\
  + glutine : boolean
};

\node[class] (uova) at (14,-15) {
  \textbf{Uova}\\[-0.3em]\rule{\linewidth}{0.2pt}\\
  + datascadenza : date
};

\node[class] (confezionati) at (17,-15) {
  \textbf{Confezionati}\\[-0.3em]\rule{\linewidth}{0.2pt}\\
  + datascadenza : date
};

% Relazioni principali - Ereditarietà
\draw[relation] (persona) -- node[midway, above left, sloped] {ISA} (cliente);
\draw[relation] (persona) -- node[midway, above right, sloped] {ISA} (dipendente);

% Relazioni Cliente
\draw[relation] (cliente) -- node[midway, left] {possiede} (tessera);
\draw[relation] (cliente) -- node[pos=0.55, above] {effettua} (ordine);

% Relazioni Dipendente
\draw[relation] (dipendente) -- node[pos=0.55, above] {gestisce} (ordine);

% Relazioni Ordine-ArticoliOrdine-Prodotto
\draw[relation] (ordine) -- node[midway, left] {contiene} (articoli);
\draw[relation] (articoli) -- node[midway, above] {riferisce} (prodotto);

% Generalizzazioni Prodotto - disposte a griglia
\draw[relation] (prodotto) -- node[midway, left] {ISA} (frutta);
\draw[relation] (prodotto) -- node[midway, left] {ISA} (verdura);
\draw[relation] (prodotto) -- node[midway, right] {ISA} (latticini);
\draw[relation] (prodotto) -- node[midway, left] {ISA} (farinacei);
\draw[relation] (prodotto) -- node[midway, left] {ISA} (uova);
\draw[relation] (prodotto) -- node[midway, right] {ISA} (confezionati);

\end{tikzpicture}
\end{center}
