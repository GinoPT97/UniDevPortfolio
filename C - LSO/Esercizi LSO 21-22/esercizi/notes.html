Esercizio 1

Reverse buffered, la dimensione del buffer va passata da command line.

Setup

Inclusione delle librerie, apertura file, ecc… Vi e’ anche la definizione di una funzione statica per invertire una stringa con lunghezza nota. 

#include <stdio.h>
#include <stdlib.h>

#include <errno.h>
#include <fcntl.h>
#include <sys/stat.h>
#include <sys/types.h>
#include <unistd.h>

#include <err.h>

static void reverse(char *buffer, int len) {
  unsigned int i = 0, j = len - 1;
  char tmp;
  while (i < j) {
    tmp = buffer[i];
    buffer[i++] = buffer[j];
    buffer[j--] = tmp;
  }
  return;
}

int main(int argc, char *argv[]) {
  int fd, n;
  char *buffer;
  unsigned int buffsize;

  char buff_input[32];

  if (argc != 3)
    err(EXIT_FAILURE,
        "Usage: %s <input_filename> <buffer size>\nthis program doesn't any "
        "check on the second parameter, please use a positive integer",
        argv[0]);

  buffsize = atoi(argv[2]);

  buffer = calloc(buffsize, sizeof(char));
  if (buffer == NULL)
    err(EXIT_FAILURE, "calloc error");

  if ((fd = open(argv[1], O_RDONLY)) == -1)
    err(EXIT_FAILURE, "open error");

Svolgimento

Il programma posiziona l’offset alla fine del file, dopodiche’ cerca di riempire tutto il buffer con una sola lettura. Dopo la lettura/riempimento del buffer viene invertito con reverse e stampato su STDOUT tramite write. L’ unico caso particolare da considerare e’ quello in cui non vi sono abbastanza byte per poter riempire il buffer, quindi quando la posizione corrente e’ inferiore alla dimensione del buffer. Tutto cio’ viene calcolato nella riga: off_t offset = cur_pos > buffsize ? cur_pos - buffsize : 0; Se e’ possibile riempire un intero buffer, l’offset viene spostato di un buffer size indietro, altrimenti viene impostato a 0, ovvero l’inizio del file. 

  off_t cur_pos = lseek(fd, -1, SEEK_END);
  if (cur_pos < 0)
    err(EXIT_FAILURE, "exit failure");

  unsigned int num_bytes_readed = 0;
  while (cur_pos > 0) {
    off_t offset = cur_pos > buffsize ? cur_pos - buffsize : 0;
    unsigned int amount_to_read = cur_pos > buffsize ? buffsize : cur_pos;
    if ((cur_pos = lseek(fd, offset, SEEK_SET)) < 0)
      err(EXIT_FAILURE, "seek failure");
    if ((num_bytes_readed = read(fd, (void *)buffer, amount_to_read)) < 0)
      err(EXIT_FAILURE, "read failure");

    reverse(buffer, amount_to_read);
    if (write(STDOUT_FILENO, (void *)buffer, num_bytes_readed) < 0)
      err(EXIT_FAILURE, "write failure");
  }

  close(fd);
  exit(0);
}

Esercizio 2

Reverse in place di un file.

Setup

La prima parte e’ pressocche’ simile al primo esercizio con qualche controllo in piu’. Vi e’ anche la definizione di due funzioni (read_frame & write_frame) che discutero’ piu’ avanti.

#include <stdio.h>

Svolgimento

L’idea di base era quella di dividere in varie porzioni/segmenti il file (a cui nel programma mi riferisco con il termine frame) e lavorare su queste porzioni piu’ semplici da gestire. Calcolo prima quanti frame vi sono (num_frame = num_byte / buffsize) e quanti byte rimangono (byte_remaining = num_byte % buffsize), che insomma non riempiono un buffer completo. Ora lo svolgimento prosegue leggendo un frame ad inizio file ed un altro alla fine, una volta letti possono essere scambiati di posto e riscritti nel file, dopo averli invertiti ovviamente. Occorre pero’ tenere conto di alcuni casi particolari:

  1. Cosa succede se il numero di frame e’ dispari?
  2. Cosa succede se non vi sono byte rimanenti?
  3. E se il numero di frame e’ divisibile per 2?

Ho trattato questi ed altri casi particolari separatamente. Prima il programma procede lavorando su frame appaiati, quelli con un corrispettivo (Risolve il problema 3). Ora se gli offset di lettura ad inizio file e fine file non si sono superati e non sono affiancati1 continua leggendo i frame rimanenti2 e byte rimanenti scrivendoli nella posizione giusta.

Sicuramente questa parte del programma potrebbe essere scritta meglio, mi ero ripromesso di sistemarla un po’ ma cosi’ facendo non condividerei mai la soluzione. Se qualcuno trova un modo piu’ elegante e snello di risolvere me lo faccia sapere! 

  off_t num_byte = info.st_size - 1; // equivale a lseek(fd_read, 0, SEEK_END)

  size_t num_bytes_readed_from_start = 0;
  size_t num_bytes_readed_from_end = 0;

  off_t num_frame = num_byte / buffsize;
  off_t byte_remaining = num_byte % buffsize;

  off_t frame_to_consume = num_frame - num_frame / 2;
  off_t frame_consumed = 0;

  if (num_byte <= buffsize) {
    num_bytes_readed_from_start = read_frame(
        fd_read, buffer_read_from_end, buffsize, current_position_from_start);
    reverse(buffer_read_from_end, num_bytes_readed_from_start);
    write_frame(fd_write, buffer_read_from_end, num_bytes_readed_from_start,
                current_position_from_start);
  } else {
    for (; frame_consumed < frame_to_consume; frame_consumed++) {
      num_bytes_readed_from_start =
          read_frame(fd_read, buffer_read_from_start, buffsize,
                     current_position_from_start);

      num_bytes_readed_from_end =
          read_frame(fd_read, buffer_read_from_end, buffsize,
                     current_position_from_end - buffsize);

      reverse(buffer_read_from_start, num_bytes_readed_from_start);
      reverse(buffer_read_from_end, num_bytes_readed_from_end);

      current_position_from_start += num_bytes_readed_from_start;
      current_position_from_end -= num_bytes_readed_from_end;

      write_frame(fd_write, buffer_read_from_end, num_bytes_readed_from_end,
                  current_position_from_start - buffsize);
      write_frame(fd_write, buffer_read_from_start, num_bytes_readed_from_start,
                  current_position_from_end);
    }

    if (current_position_from_start < current_position_from_end - 1) {
      off_t from_where = current_position_from_start;
      if (frame_consumed < frame_to_consume) {
        num_bytes_readed_from_start =
            read_frame(fd_read, buffer_read_from_start, buffsize,
                       current_position_from_start);
        from_where = current_position_from_start + num_bytes_readed_from_start;
        reverse(buffer_read_from_start, num_bytes_readed_from_start);
      }
      num_bytes_readed_from_end =
          read_frame(fd_read, buffer_read_from_end, byte_remaining, from_where);
      reverse(buffer_read_from_end, num_bytes_readed_from_end);
      write_frame(fd_write, buffer_read_from_end, num_bytes_readed_from_end,
                  current_position_from_start);
      current_position_from_start += num_bytes_readed_from_end;
      if (frame_consumed < frame_to_consume) {
        write_frame(fd_write, buffer_read_from_start,
                    num_bytes_readed_from_start, current_position_from_start);
      }
    }
  }
}

Funzioni read/write frame

Queste due funzioni (duali tra loro), servono a posizionare l’offset (in questo caso e’ meglio immaginarlo come un cursore) nella posizione indicata da from e leggono/scrivono buffsize byte da/nel buffer. Si puo’ anche interpretare in questo modo (faccio un esempio solo per read_frame): Legge, da fd, ed inserisce in buffer un numero pari a buffsize byte dalla posizione from. 

size_t read_frame(int fd, char *buffer, size_t buffsize, off_t from) {
  if (lseek(fd, from, SEEK_SET) < 0)
    err(EXIT_FAILURE, "seek read failure");
  size_t num_byte_readed = 0;
  if ((num_byte_readed = read(fd, buffer, buffsize)) < 0)
    err(EXIT_FAILURE, "read frame failure");
  return num_byte_readed;
}

size_t write_frame(int fd, char *buffer, size_t buffsize, off_t from) {
  if (lseek(fd, from, SEEK_SET) < 0)
    err(EXIT_FAILURE, "seek write failure");
  size_t num_byte_written = 0;
  if ((num_byte_written = write(fd, buffer, buffsize)) < 0)
    err(EXIT_FAILURE, "write frame failure");
  return num_byte_written;
}

Esercizio 3

Si implementi un programma C che prende come argomenti tre nomi di files f1, f2 , f3 (f1 deve essere un file esistente) ed il nome di un eseguibile (es. rev.out) che accetta a sua volta il path di un file e lo stampa in ordine inverso di bytes sullo standard output. Il programma genera un primo figlio che esegue rev.out su f1 e scrive l’output in f2. Successivamente, il programma genera un secondo figlio che che esegue rev.out su f2 e scrive a sua volta l’output in f3.

Ridirigo lo STDOUT sul primo file in output, eseguo il programma, dopodiche’ ripeto il procedimento ridirigendo di nuovo lo STDOUT sul secondo file e rieseguendo il programma usando come input il primo output.

Accorgimenti

Invece che usare execl avrei preferito usare execlp, questo perche’ con le variabili di ambiente potevo modificare la variabile di $PATH per cercare l’eseguibile del programma. Infatti, per come e’ scritto ora, per passare l’eseguibile devo inserire tutto il path assoluto, nel mio caso /home/anto/Uni/lso/exerecise/esercizi_25-10-2021/reverse_buffered. Alla fine non ho usato execlp perche’ anche modificando le variabili di ambiente l’eseguibile non veniva trovato, probabilmento dovevo fare qualche passaggio in piu’ tipo:

  1. Estrarre il valore associato a PATH.
  2. Concatenare a PATH il nome dell’eseguibile.
  3. Utilizzare l’eseguibile.

Se modifico la variabile $PATH (i.e. export PATH="$PATH:/percorso/per/l'eseguibile") prima dell’esecuzione il programma funziona anche con execlp.
