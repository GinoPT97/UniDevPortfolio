Traccia

  1. realizzare un programma che accetta da riga di comando due numeri interi n ed m, e crea n produttori ed m consumatori.
  2. produttori e consumatori condividono un array di 100 interi.
  3. ogni produttore aspetta un numero casuale di secondi tra 1 e 10, e poi produce (cioe’ inserisce nell’array) da 1 a 5 numeri casuali. Se il produttore trova l’array pieno, salta il turno.
  4. ogni consumatore aspetta che ci sia un numero da consumare, e poi stampa a video il proprio tid e il valore consumato.

Include, definizioni struct e definizioni funzioni

#define ARRAY_SIZE 100

typedef struct thread_arg {
  int num_item;
  int cur_pos_read;
  int cur_pos_write;
  int array[ARRAY_SIZE];
  pthread_mutex_t arg_mutex;
  int producers_terminated;
} * thread_arg;

void *produce_job(void *param);
void *consume_job(void *param);

int check_input(char *s);

Main

int main(int argc, char *argv[]) {
  if (argc != 3) {
    printf("Usage: %s n m\n", argv[0]);
    exit(EXIT_FAILURE);
  }
  if (check_input(argv[1])) {
    printf("%s is not an integer\n", argv[1]);
    exit(EXIT_FAILURE);
  }
  if (check_input(argv[2])) {
    printf("%s is not an integer\n", argv[2]);
    exit(EXIT_FAILURE);
  }
  int n = atoi(argv[1]);
  int m = atoi(argv[2]);

  if (!(n > 0 && m > 0)) {
    printf("hey you need at least one producer and one consumer!\n");
    exit(EXIT_FAILURE);
  }

  struct thread_arg arg = {0};
  int err_init_mutex;
  if ((err_init_mutex = pthread_mutex_init(&arg.arg_mutex, NULL))) {
    printf("Error init struct arg mutex: %s\n", strerror(err_init_mutex));
    exit(EXIT_FAILURE);
  }

  pthread_t *producers = (pthread_t *)calloc(n, sizeof(pthread_t));
  pthread_t *consumers = (pthread_t *)calloc(m, sizeof(pthread_t));

  int err_start_thread;
  for (int i = 0; i < n; i++) {
    if ((err_start_thread =
             pthread_create(&producers[i], NULL, produce_job, (void *)&arg))) {
      printf("Error starting producer thread [%d]: %s\n", i,
             strerror(err_start_thread));
      exit(EXIT_FAILURE);
    }
  }
  for (int i = 0; i < m; i++) {
    if ((err_start_thread =
             pthread_create(&consumers[i], NULL, consume_job, (void *)&arg))) {
      printf("Error starting consumer thread [%d]: %s\n", i,
             strerror(err_start_thread));
      exit(EXIT_FAILURE);
    }
  }

  int err_join_thread;
  for (int i = 0; i < n; i++) {
    if ((err_join_thread = pthread_join(producers[i], NULL))) {
      printf("Error joining producer thread [%d]: %s\n", i,
             strerror(err_join_thread));
      exit(EXIT_FAILURE);
    }
  }
  pthread_mutex_lock(&arg.arg_mutex);
  printf("producers terminated\n");
  arg.producers_terminated = 1;
  pthread_mutex_unlock(&arg.arg_mutex);

  for (int i = 0; i < m; i++) {
    if ((err_join_thread = pthread_join(consumers[i], NULL))) {
      printf("Error joining consumer thread [%d]: %s\n", i,
             strerror(err_join_thread));
      exit(EXIT_FAILURE);
    }
  }

  int err_destroy;
  if ((err_destroy = pthread_mutex_destroy(&arg.arg_mutex))) {
    printf("Error destroying struct arg mutex: %s\n", strerror(err_destroy));
    exit(EXIT_FAILURE);
  }

  free(producers);
  free(consumers);

  return 0;
}

Funzione da associare ai produttori

void *produce_job(void *param) {
  thread_arg arg = (thread_arg)param;
  srand(time(NULL));
  sleep(rand() % 10 + 1);
  int rand_iter = rand() % 5;
  printf("tid: %lu\twill produce (at most) %d value\n", pthread_self(),
         rand_iter + 1);
  int i = 0;
  while (i <= rand_iter) {
    if (pthread_mutex_trylock(&arg->arg_mutex) == 0) {
      if (arg->num_item != ARRAY_SIZE) {
        arg->array[arg->cur_pos_write] = rand();
        arg->cur_pos_write = (arg->cur_pos_write + 1) % ARRAY_SIZE;
        arg->num_item++;
        i++;
      }
      pthread_mutex_unlock(&arg->arg_mutex);
    } else { /* EBUSY*/
    }
  }
  return NULL;
}

Funzione da associare ai consumatori

void *consume_job(void *param) {
  thread_arg arg = (thread_arg)param;
  while (1) {
    if (pthread_mutex_trylock(&arg->arg_mutex) == 0) {
      if (arg->num_item != 0) {
        printf("tid: %lu\tconsumed:%d\n", pthread_self(),
               arg->array[arg->cur_pos_read]);
        arg->cur_pos_read = (arg->cur_pos_read + 1) % ARRAY_SIZE;
        arg->num_item--;
      }
      if (arg->num_item == 0 && arg->producers_terminated) {
        pthread_mutex_unlock(&arg->arg_mutex);
        break;
      }
      pthread_mutex_unlock(&arg->arg_mutex);
    } else {
      // wait for producer to do some job
      // sleep(1); // just to test different behaviour
    }
  }
  return NULL;
}

Considerazioni

Traccia ambigua (in realta’ non troppo, ma non si sa mai)

Nella traccia, in particolare al punto 4, potrei non aver capito bene. Io l’ho interpretata nel seguente modo: “se l’array e’ pieno il thread produttore aspetta che un qualche consumatore consumi un valore e successivamente produce un nuovo valore”. Un’ altra interpretazione e’: “se l’array e’ pieno il thread non produce il valore e va avanti, producendo quindi un valore in meno”. La prima mi sembra piu’ sensata, quindi ho implemtato quella.

Utilizzo dei lock

Avevo intezione di scrivere tre versioni diverse di questo programma, usando in ognuna diversi approcci rispettivamente: solo con pthread_mutex_lock, utilizzare anche pthread_mutex_trylock ed infine pthread_cond_wait. Il problema e’ che con solo la funzione pthread_mutex_lock non riuscivo ad evitare deadlock. A volte il programma non era neanche in un deadlock effettivo, nel senso che dopo un po’ si sbloccava e terminava, ma i tempi di esecuzione (anche con pochissimi produttori e consumatori, tipo 2 produttori ed 2 consumatori) erano improponibili.

Perche’ vi erano deadlock con pthread_mutex_lock

  • Un po’ id contesto

    Per dare un po’ di contesto, il consumatore era scritto, piu’ o meno, nel seguente modo:

    while(1){
    // stuff here
    phtread_mutex_lock(&lock);
    if(non_ci_sono_valori_da_consumare){
        pthread_mutex_unlock(&lock);
        continue; // salta il ciclo
    }else{
        // stuff here
        consuma_un_valore;
    }
    pthread_mutex_unlock(&lock);
}

    Come accenato sopra, non vi erano degli effettivi deadlock, durante alcune esecuzioni e’ capitato che dopo 15/20 minuti (il mio portatile stava decollando) l’esecuzione riprendesse ed andasse a buon fine. Per capire un po’ come si comportasse il codice, ho inserito delle stampe nei consumatori (in particolare nell’if di cui sopra, prima del continue). Quello che ho osservato e’ che differenti consumatori riuscivano a scrivere sullo STDOUT, quindi non vi era un lock bloccato che non veniva rilasciato, ma piuttosto erano i produttori che non riuscivano mai a prendere il controllo del lock. Infatti inserendo delle printf nei produttori, queste non venivano mai stampate a video.

  • Spiegazione

    Quello che scrivo ora sono le mie supposizioni personali, non sono riuscito a trovare articoli, post su stackoverflow et similia che possano confermare le mie ipotesi. (Anche perche’ non credo di aver posto bene la domanda, infatti mi uscivano molti risulatati correlati a problemi di deadlock effettivo, beh almeno Google non mi ha risposto 42).

    Dato che i consumatori iniziavano la loro esecuzione prima dei produttori (nel senso che i produttori chiamavano subito una sleep), venivano “schedulati” per primi, quando ottenevano il lock controllavano che non c’erano valori da consumare, rilasciavano il lock e ripetevano il ciclo daccapo.

    Nel momento in cui i produttori terminavano i relativi sleep, cercavano di acquisire il lock e per poter fare cio’ occorreva aspettare che venissero schedulati e che il lock fosse libero. (Da altri test che ho fatto, le solite printf inserite nel codice, in particolare prima di acquisire il lock, i produttori non stampavano messaggi.) Il problema e’ che lo scheduler non selezionava mai (o quasi, infatti a volte si “sbloccava”). Leggendo dal libro di testo Sistemi operativi: concetti ed esempi, Linux solitamente usa uno scheduler chiamato completely fair (che e’ quello che usa la mia macchina su cui esegue il codice) che tende a favorire, in base ad alcuni parametri, ovvero i nice value (si possono vedere anche in top o htop, anche se relativi al processo e non ai thread di un processo), processi e thread che occupano poco tempo di elaborazione. Una volta che il processo viene schedulato deve a sua volta scegliere tra quali dei suoi thread schedulare, e tra i produttori ed i consumatori favoriva sempre quest’ultimi.

  • Soluzione

    Problemi del genere possono accadere e per poter essere evitati occorre implementare un meccanismo di aging cosi’ che anche i thread meno favoriti possano essere schedulati. Nella libreria pthread.h ho trovato alcune funzioni per modificare la priorita’ di scheduling (tra cui pthread_setschedprio), ma non mi sono azzardato ad usarle.

  • Conclusione

    Questa e’ la spiegazione che mi son dato, ma molto piu probabilmente potrei aver malinterpretato alcuni comportamenti ed aver utilizzato parecchia fantasia. L’unica cosa che mi fa pensare che potrei aver individuato il problema (o uno dei problemi) e’ che il programma si sia “sbloccato” in modo autonomo. E’ accaduto una volta per sbaglio perche’ non avevo interrotto l’esecuzione, e in altri 5 tentativi l’evento si e’ ripetuto una sola volta. Nelle rimanenti 4 esecuzioni, l’ho interrotto dopo circa 30 minuti.
